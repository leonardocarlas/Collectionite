    function language_to_id($lingua)
    {
        $id_language = 0;
        if($lingua == "English")
        {
            $id_language = 1 ;
        }
        if($lingua == "French")
        {
            $id_language = 2;
        }
        if($lingua == "German")
        {
            $id_language = 3;
        }
        if($lingua == "Spanish")
        {
            $id_language = 4;
        }
        if($lingua == "Italian")
        {
            $id_language = 5;
        }
        if($lingua == "Simplified Chinese")
        {
            $id_language = 6;
        }
        if($lingua == "Japanese")
        {
            $id_language = 7;
        }
        if($lingua == "Portuguese")
        {
            $id_language = 8;
        }
        if($lingua == "Russian")
        {
            $id_language = 9;
        }
        if($lingua == "Korean")
        {
            $id_language = 10;
        }
        if($lingua == "Traditional Chinese")
        {
            $id_language = 11;
        }
        return $id_language;
    }

function condition_to_id($condizione)
{
        $id_conditions = 0;
        switch ($condizione) {
            case "M":
                $id_conditions = 1 ;
                break;
            case "NM":
                $id_conditions = 2 ;
                break;
            case "EX":
                $id_conditions = 3 ;
                break;
            case "GD":
                $id_conditions = 4 ;
                break;
            case "LP":
                $id_conditions = 5 ;
                break;
            case "PL":
                $id_conditions = 6 ;
                break;
            case "P":
                $id_conditions = 7 ;
                break;
        }
        return $id_conditions;
}
